CREATE TABLE `cms_user` (
 `id` int(11) NOT NULL AUTO_INCREMENT,
 `username` varchar(30) NOT NULL,
 `password` char(32) NOT NULL,
 `email` varchar(100) DEFAULT NULL,
 `birthday` varchar(30) NOT NULL,

 )ENGINE=InnoDB;
